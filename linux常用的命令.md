```shell
pwd       查看当前路径位置
ifconfig  查看网络设备信息
free -h                 检查服务器内存
df -lh                  查看磁盘空间
mkdir                   新建目录
cd                      切换目录
du -sh *                查看当前文件夹下文件占用空间
cat                     查看文件
usname -a               查看系统版本
which                   查找路径
wget                    下载压缩包
tar -zxvf               解压  
find -name              查找
cp                      复制
nohop java -jar *.jar & 后台启动程序
jobs                    查看当前运行任务
ps -ef                  查看当前进程
ps -ef | grep 'java'    通过管道和grep筛选
netstat -ntlp           查看端口
tail -n 10 error.log    查看错误日志最新10行
sz 文件                  将文件下载到本地4
kill -9 %l              杀死老进程
```
